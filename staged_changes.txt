+class ImageAttachment(BaseModel):
+    type: Literal["image"] = "image"
+    source: FileSource

+class DocumentAttachment(BaseModel):
+    type: Literal["document"] = "document"
+    source: FileSource

+class AudioAttachment(BaseModel):

+    source: FileSource
+
+class VideoAttachment(BaseModel):
